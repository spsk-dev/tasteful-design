**2. Create `evals/parse-review-output.sh`** -- sourced by the runner, provides extraction functions:

```bash
#!/usr/bin/env bash
# SpSk Output Parser -- extracts scores and verdicts from design-review terminal output
# Sourced by run-quality-evals.sh. Not run directly.

extract_verdict() {
  local output="$1"
  # Match "Verdict: SHIP", "Verdict: CONDITIONAL SHIP", "Verdict: CONDITIONAL", "Verdict: BLOCK"
  # Anchor to **Verdict: prefix to avoid matching stray words in review text
  echo "$output" | grep -oE '\*\*Verdict: (SHIP|CONDITIONAL SHIP|CONDITIONAL|BLOCK)' | head -1 | sed 's/\*\*Verdict: //'
}

extract_overall() {
  local output="$1"
  # Match "Score: X.XX/4.0" from the boss output header
  echo "$output" | grep -oE '\*\*Score: [0-9]+\.[0-9]+/4\.0' | head -1 | grep -oE '[0-9]+\.[0-9]+' | head -1
}

extract_score() {
  local dimension="$1"
  local output="$2"
  # Map dimension names to specialist table row labels
  local label
  case "$dimension" in
    intent_match)  label="Intent Match" ;;
    originality)   label="Originality" ;;
    ux_flow)       label="UX Flow" ;;
    typography)    label="Typography" ;;
    color)         label="Color" ;;
    layout)        label="Layout" ;;
    icons)         label="Icons" ;;
    motion)        label="Motion" ;;
    copy)          label="Copy" ;;
    code_a11y)     label="Code" ;;
    *)             label="$dimension" ;;
  esac
  # Extract N from "| Label | N/4 | Wx |" in the scores table
  echo "$output" | grep -i "| $label " | grep -oE '[0-9]+\.?[0-9]*/4' | head -1 | grep -oE '^[0-9]+\.?[0-9]*'
}

float_in_range() {
  local value="$1" min="$2" max="$3"
  python3 -c "import sys; v,lo,hi=float('$value'),float('$min'),float('$max'); sys.exit(0 if lo<=v<=hi else 1)"
}
```

**Section 0: Setup and dependencies**
```bash
#!/usr/bin/env bash
set -uo pipefail
# Note: set -e intentionally omitted -- check() handles errors (same as validate-structure.sh)

SCRIPT_DIR="$(cd "$(dirname "${BASH_SOURCE[0]}")" && pwd)"
REPO_ROOT="$(cd "$SCRIPT_DIR/.." && pwd)"

# Source the output parser
source "$SCRIPT_DIR/parse-review-output.sh"

PASS=0
FAIL=0
SKIP=0

check() {
  local label="$1"
  shift
  if "$@" >/dev/null 2>&1; then
    echo "[PASS] $label"
    PASS=$((PASS + 1))
  else
    echo "[FAIL] $label"
    FAIL=$((FAIL + 1))
  fi
}

check_skip() {
  local label="$1"
  local reason="$2"
  echo "[SKIP] $label ($reason)"
  SKIP=$((SKIP + 1))
}
```

**Section 3: Review invocation function**
Create a `run_review()` function that invokes design-review via `claude -p`:

```bash
run_review() {
  local fixture_url="$1"
  local output_file="$2"

  echo "[INFO] Invoking design-review on $fixture_url ..."
  echo "[INFO] This may take 5-15 minutes per fixture."

  # Strategy A (preferred): Task description with plugin-dir
  claude -p \
    --plugin-dir "$REPO_ROOT" \
    --dangerously-skip-permissions \
    --model sonnet \
    --max-turns 30 \
    --output-format text \
    "Run a full design review on $fixture_url. Use the /design-review command. Output the complete review with specialist scores table and verdict." \
    2>/dev/null > "$output_file"

  local exit_code=$?
  if [ $exit_code -ne 0 ]; then
    echo "[WARN] claude -p exited with code $exit_code"
  fi

  # Validate we got a real review (not empty or error)
  if [ ! -s "$output_file" ]; then
    echo "[ERROR] Empty output from claude -p"
    return 1
  fi

  # Check for verdict presence as smoke test
  local verdict
  verdict=$(extract_verdict "$(cat "$output_file")")
  if [ -z "$verdict" ]; then
    echo "[WARN] No verdict found in output. Review may have failed."
    echo "[DEBUG] First 500 chars of output:"
    head -c 500 "$output_file"
    return 1
  fi

  echo "[INFO] Review complete. Verdict: $verdict"
  return 0
}
```

**Important:** If `--dry-run` is active, `run_review()` should skip the claude invocation and look for `evals/results/cache-{fixture_name}.txt` instead. If the cache file does not exist, skip that fixture's assertions.
